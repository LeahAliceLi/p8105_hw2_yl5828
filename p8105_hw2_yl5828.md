pols_month_df =
  read_csv("./fivethirtyeight_datasets/pols-month.csv", na = c("NA", ".", "")) %>%

  separate(mon, into = c("year", "month", "day"), sep = "-") %>%
  mutate(

  select(-prez_dem, -prez_gop, -day)

snp_df =
